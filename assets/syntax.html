  1. Usually statements evaluated from text-head to text-end.
  2. if
    if(condition){ then-statements } else { else-statements };
    -- if 'conditon' is true, then evaluate 'then-statements' else 'else-statements'.
    -- after 'else' is optional.
  3. while
    while(condition){ statements };
    -- while 'condition' is true, then evaluate statements.
  4. built-in operators
    +, -, *, /, <, <=, >, >=
  5. define variable
    def variableName;
    def variableName = initialValue;
  6. define method
    fun methodName(){ statements };
    fun methodName(arg1, arg2, ..., argN){ statements };
  7. call method
    methodName();
    methodName(arg1, arg2, ..., argN){ statements };
  8. object
    object { 'define valiable' or 'define method' list }; -- ex.
    def obj = object { def a = 1; def b = 2; fun f(n){ n + a; }; fun g(n){ n * b; }; };
  9. call object's method
    objectName.methodName();
    objectName.methodName(arg1, arg2, ..., argN);
  10. clone object
    new objectName;
    -- ex. def obj2 = new objectName;
  11. multiple derived object
    objectName1 + objectName2;
    -- ex. def obj3 = obj1 + obj2;
